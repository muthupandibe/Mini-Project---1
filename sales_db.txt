--Branches Table Creation 
CREATE TABLE branches (
    branch_id SERIAL PRIMARY KEY,
    branch_name VARCHAR(100) NOT NULL,
    branch_admin_name VARCHAR(100) NOT NULL
);

--users Table Creation 
CREATE TABLE users (
    user_id SERIAL PRIMARY KEY,
    username VARCHAR(100) UNIQUE NOT NULL,
    password VARCHAR(100) NOT NULL,
    email VARCHAR(255) UNIQUE NOT NULL,
    branch_id INT,
    role VARCHAR(20) NOT NULL
        CHECK (role IN ('Super Admin', 'Admin')),
    FOREIGN KEY (branch_id)
        REFERENCES branches(branch_id)
);


--customer_sales Table Creation 
CREATE TABLE customer_sales (
    sale_id SERIAL PRIMARY KEY,

    branch_id INT NOT NULL,

    sale_date DATE NOT NULL,

    customer_name VARCHAR(100),

    mobile_number VARCHAR(15) UNIQUE,

    product_name VARCHAR(30)
        CHECK (product_name IN ('DS', 'DA', 'BA', 'FSD')),

    gross_sales NUMERIC(12,2) NOT NULL,

    received_amount NUMERIC(12,2) NOT NULL DEFAULT 0,

    pending_amount NUMERIC(12,2)
        GENERATED ALWAYS AS
        (gross_sales - received_amount) STORED,

    status VARCHAR(10) NOT NULL DEFAULT 'Open'
        CHECK (status IN ('Open', 'Close')),

    CONSTRAINT fk_customer_sales_branch
        FOREIGN KEY (branch_id)
        REFERENCES branches(branch_id)
);

--payment_splits Table Creation 

CREATE TABLE payment_splits (
    payment_id SERIAL PRIMARY KEY,

    sale_id INT NOT NULL,

    payment_date DATE NOT NULL DEFAULT CURRENT_DATE,

    amount_paid NUMERIC(12,2) NOT NULL
        CHECK (amount_paid > 0),

    payment_method VARCHAR(50) NOT NULL
        CHECK (payment_method IN ('Cash', 'UPI', 'Card')),

    CONSTRAINT fk_payment_sale
        FOREIGN KEY (sale_id)
        REFERENCES customer_sales(sale_id)
        ON DELETE CASCADE
);

--AUTOMATION LOGIC (TRIGGERS)
CREATE OR REPLACE FUNCTION update_received_amount()
RETURNS TRIGGER
LANGUAGE plpgsql
AS $$
BEGIN

    UPDATE customer_sales
    SET received_amount =
        COALESCE(
            (
                SELECT SUM(amount_paid)
                FROM payment_splits
                WHERE sale_id = NEW.sale_id
            ),
            0
        )
    WHERE sale_id = NEW.sale_id;

    UPDATE customer_sales
    SET status =
        CASE
            WHEN (gross_sales - received_amount) <= 0
                THEN 'Close'
            ELSE 'Open'
        END
    WHERE sale_id = NEW.sale_id;

    RETURN NEW;

END;
$$;
CREATE TRIGGER trg_update_received_amount
AFTER INSERT OR UPDATE OR DELETE
ON payment_splits
FOR EACH ROW
EXECUTE FUNCTION update_received_amount();

--3.Join-Based Queries 
--Joins
--Retrieve sales details along with the branch name
SELECT
    cs.sale_id,
    cs.customer_name,
    cs.mobile_number,
    cs.product_name,
    cs.gross_sales,
    b.branch_name
FROM customer_sales cs
JOIN branches b
    ON cs.branch_id = b.branch_id;

--Retrieve sales details along with total payment received (using payment_splits).
SELECT
    cs.sale_id,
    cs.customer_name,
    cs.gross_sales,
    COALESCE(SUM(ps.amount_paid), 0) AS total_payment_received
FROM customer_sales cs
LEFT JOIN payment_splits ps
    ON cs.sale_id = ps.sale_id
GROUP BY
    cs.sale_id,
    cs.customer_name,
    cs.gross_sales;

--Show branch-wise total gross sales (using JOIN & GROUP BY)
SELECT
    b.branch_name,
    SUM(cs.gross_sales) AS total_gross_sales
FROM branches b
JOIN customer_sales cs
    ON b.branch_id = cs.branch_id
GROUP BY
    b.branch_name
ORDER BY
    total_gross_sales DESC;

--Financial Tracking Queries
--1. Find sales where the pending amount is greater than 5000
SELECT
    sale_id,
    customer_name,
    mobile_number,
    gross_sales,
    received_amount,
    pending_amount
FROM customer_sales
WHERE pending_amount > 5000;

--2. Retrieve Top 3 Highest Gross Sales
SELECT
    sale_id,
    customer_name,
    gross_sales
FROM customer_sales
ORDER BY gross_sales DESC
LIMIT 3;

--3. Find the Branch with Highest Total Gross Sales
SELECT
    b.branch_name,
    SUM(cs.gross_sales) AS total_gross_sales
FROM branches b
JOIN customer_sales cs
    ON b.branch_id = cs.branch_id
GROUP BY b.branch_name
ORDER BY total_gross_sales DESC
LIMIT 1;

--4. Retrieve Monthly Sales Summary (Month & Year Wise)
SELECT
    TO_CHAR(sale_date, 'YYYY-MM') AS month_year,
    COUNT(*) AS total_sales,
    SUM(gross_sales) AS total_gross_sales,
    SUM(received_amount) AS total_received,
    SUM(pending_amount) AS total_pending
FROM customer_sales
GROUP BY TO_CHAR(sale_date, 'YYYY-MM')
ORDER BY month_year;

--5. Calculate Payment Method-wise Total Collection
SELECT
    payment_method,
    SUM(amount_paid) AS total_collection
FROM payment_splits
GROUP BY payment_method
ORDER BY total_collection DESC;
