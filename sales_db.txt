-- Branches Table Creation 
CREATE TABLE branches (
    branch_id SERIAL PRIMARY KEY, -- Unique ID for each branch
    branch_name VARCHAR(100) NOT NULL, -- Name of the branch
    branch_admin_name VARCHAR(100) NOT NULL -- Admin responsible for the branch
);

-- users Table Creation 
CREATE TABLE users (
    user_id SERIAL PRIMARY KEY, -- Unique ID for the user
    username VARCHAR(100) UNIQUE NOT NULL, -- Unique login username
    password VARCHAR(100) NOT NULL, -- Encrypted or plain password
    email VARCHAR(255) UNIQUE NOT NULL, -- Unique user email
    branch_id INT, -- Link to the branches table
    role VARCHAR(20) NOT NULL -- User role identifier
        CHECK (role IN ('Super Admin', 'Admin')), -- Validate role type
    FOREIGN KEY (branch_id) -- Link to branch
        REFERENCES branches(branch_id) -- Reference key
);

-- customer_sales Table Creation 
CREATE TABLE customer_sales (
    sale_id SERIAL PRIMARY KEY, -- Unique ID for each sale
    branch_id INT NOT NULL, -- ID of branch where sale occurred
    sale_date DATE NOT NULL, -- Date of the sale
    customer_name VARCHAR(100), -- Customer full name
    mobile_number VARCHAR(15) UNIQUE, -- Unique customer contact
    product_name VARCHAR(30) -- Product identifier
        CHECK (product_name IN ('DS', 'DA', 'BA', 'FSD')), -- Allowed products
    gross_sales NUMERIC(12,2) NOT NULL, -- Total price of product
    received_amount NUMERIC(12,2) NOT NULL DEFAULT 0, -- Total money collected
    pending_amount NUMERIC(12,2) -- Remaining balance
        GENERATED ALWAYS AS (gross_sales - received_amount) STORED, -- Computed column
    status VARCHAR(10) NOT NULL DEFAULT 'Open' -- Payment status
        CHECK (status IN ('Open', 'Close')), -- Only Open or Close allowed
    CONSTRAINT fk_customer_sales_branch -- Naming the foreign key
        FOREIGN KEY (branch_id) -- Linking to branches
        REFERENCES branches(branch_id) -- Reference column
);

-- payment_splits Table Creation 
CREATE TABLE payment_splits (
    payment_id SERIAL PRIMARY KEY, -- Unique ID for payment
    sale_id INT NOT NULL, -- Link to original sale
    payment_date DATE NOT NULL DEFAULT CURRENT_DATE, -- Date payment was made
    amount_paid NUMERIC(12,2) NOT NULL -- Amount received
        CHECK (amount_paid > 0), -- Ensure payment is positive
    payment_method VARCHAR(50) NOT NULL -- Cash, UPI, or Card
        CHECK (payment_method IN ('Cash', 'UPI', 'Card')), -- Restrict input
    CONSTRAINT fk_payment_sale -- Constraint name
        FOREIGN KEY (sale_id) -- Foreign key to sale
        REFERENCES customer_sales(sale_id) -- Referenced column
        ON DELETE CASCADE -- Remove splits if sale is deleted
);

-- AUTOMATION LOGIC (TRIGGERS)
CREATE OR REPLACE FUNCTION update_received_amount() -- Create trigger function
RETURNS TRIGGER -- Return type
LANGUAGE plpgsql -- Using PL/pgSQL
AS $$ -- Start function body
BEGIN -- Begin logic
    UPDATE customer_sales -- Update the main sale table
    SET received_amount = -- Set the sum of payments
        COALESCE( -- Handle nulls as 0
            (
                SELECT SUM(amount_paid) -- Get total paid for this sale
                FROM payment_splits
                WHERE sale_id = NEW.sale_id
            ), 0
        )
    WHERE sale_id = NEW.sale_id; -- Identify correct record

    UPDATE customer_sales -- Second update for status
    SET status = -- Change status based on balance
        CASE
            WHEN (gross_sales - received_amount) <= 0 -- If paid in full
                THEN 'Close' -- Mark as closed
            ELSE 'Open' -- Keep as open
        END
    WHERE sale_id = NEW.sale_id; -- Identify correct record

    RETURN NEW; -- Return the updated row
END; -- End function
$$; -- End block
